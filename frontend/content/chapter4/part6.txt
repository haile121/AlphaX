8 Develop a program that uses the function factorial() of exercise 6 to compute anapproximation of e (Euler’s number). Base your approximation on the followingformula for e:1 + 1/1! + 1/2! + 1/3! + …

Full code

#include <iostream>

#include <iomanip> // For std::setprecision

// Function to calculate factorial recursively

long long factorial(int n) {

    if (n < 0) {

        return 0; // Factorial is not defined for negative numbers

    }

    if (n == 0 || n == 1) {

        return 1;

    }

    return n * factorial(n - 1);

}

int main() {

    double e_approximation = 0.0;

    int num_terms;

    std::cout << "Enter the number of terms to use for approximating e: ";

    std::cin >> num_terms;

    if (num_terms <= 0) {

        std::cout << "Number of terms must be positive." << std::endl;

        return 1;

    }



    for (int i = 0; i < num_terms; ++i) {

        e_approximation += 1.0 / factorial(i);

    }

    std::cout << std::fixed << std::setprecision(15); // Set precision for output

    std::cout << "Approximation of e using " << num_terms << " terms: " << e_approximation << std::endl;



    return 0;

}

bool isEven(int number) {

    return (number % 2 == 0);
